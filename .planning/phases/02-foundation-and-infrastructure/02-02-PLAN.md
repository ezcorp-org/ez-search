  <files>

  <action>
    Create `src/services/file-scanner.ts` that exports an `scanFiles` async generator function.

    Implementation details:

    1. Import types from `../types.js`: `ScannedFile`, `ScanOptions`, `EXTENSION_MAP`, `BUILTIN_EXCLUSIONS`
    2. Import `ignore` from 'ignore' package
    3. Import `fs` (readFileSync, existsSync) and `fs/promises` (opendir, stat), `path`

    4. `scanFiles(rootDir: string, opts: ScanOptions): AsyncGenerator<ScannedFile>`:
       - Resolve rootDir to absolute path via `path.resolve(rootDir)`
       - Create an `ignore` instance, always add BUILTIN_EXCLUSIONS
       - If `opts.useIgnoreFiles` is true, read `.gitignore` and `.cursorignore` from rootDir (if they exist) and add their contents to the ignore instance
       - Delegate to internal `walkDir` generator

    5. Internal `walkDir(dir, rootDir, ig, opts)` async generator:
       - Use `await fsp.opendir(dir)` to iterate entries
       - For each entry:
         - Compute `fullPath = path.join(dir, entry.name)` and `relPath = path.relative(rootDir, fullPath)`
         - Skip symlinks (`entry.isSymbolicLink()`)
         - For directories: check `ig.ignores(relPath + '/')` AND `ig.ignores(relPath)` — if either matches, skip. Otherwise recurse.
         - For files: check `ig.ignores(relPath)` — if matches, skip. Look up extension in EXTENSION_MAP. If unknown extension, skip file. If `opts.typeFilter` set and doesn't match, skip. Get file size via `fsp.stat(fullPath)`. Yield `ScannedFile`.

    Critical implementation notes (from research):
    - `ignore` requires relative paths with NO leading './' — `path.relative()` handles this correctly
    - Check both `relPath + '/'` and `relPath` for directories (gitignore trailing slash semantics)
    - Skip symlinks entirely (no cycle detection needed)
    - Extension lookup should be case-insensitive (`.toLowerCase()`)
  </action>

    import { scanFiles } from './src/services/file-scanner.js';
    import { mkdirSync, writeFileSync, rmSync } from 'fs';
    import { join } from 'path';

    // Test 1: Scan with ignore files enabled
    const files = [];
    for await (const f of scanFiles('.', { useIgnoreFiles: true })) {
      files.push(f);
    }
    console.log(`Found ${files.length} files with ignores`);
    const hasNodeModules = files.some(f => f.relativePath.startsWith('node_modules'));
    console.log(`Has node_modules: ${hasNodeModules} (should be false)`);

    // Test 2: Scan with type filter
    const codeFiles = [];
    for await (const f of scanFiles('.', { useIgnoreFiles: true, typeFilter: 'code' })) {
      codeFiles.push(f);
    }
    console.log(`Found ${codeFiles.length} code files`);
    const allCode = codeFiles.every(f => f.type === 'code');
    console.log(`All code type: ${allCode} (should be true)`);

    // Test 3: Scan without ignore files
    const noIgnoreFiles = [];
    for await (const f of scanFiles('.', { useIgnoreFiles: false })) {
      noIgnoreFiles.push(f);
    }
    console.log(`Found ${noIgnoreFiles.length} files without ignore files`);
    const noIgnoreHasNodeModules = noIgnoreFiles.some(f => f.relativePath.startsWith('node_modules'));
    console.log(`No-ignore has node_modules: ${noIgnoreHasNodeModules} (should be false — built-in exclusion)`);

    // Test 4: .cursorignore-specific verification
    // Create a temp directory with both .gitignore and .cursorignore
    const tmpDir = `/tmp/ez-scanner-cursorignore-test-${Date.now()}`;
    mkdirSync(join(tmpDir, 'src'), { recursive: true });
    writeFileSync(join(tmpDir, 'src', 'keep.ts'), 'export const x = 1;');
    writeFileSync(join(tmpDir, 'src', 'secret.ts'), 'export const key = "secret";');
    writeFileSync(join(tmpDir, 'src', 'notes.md'), '# Notes');
    writeFileSync(join(tmpDir, '.gitignore'), '# nothing ignored by git\n');
    writeFileSync(join(tmpDir, '.cursorignore'), 'src/secret.ts\n');

    const cursorFiles = [];
    for await (const f of scanFiles(tmpDir, { useIgnoreFiles: true })) {
      cursorFiles.push(f);
    }
    const hasSecret = cursorFiles.some(f => f.relativePath === 'src/secret.ts');
    const hasKeep = cursorFiles.some(f => f.relativePath === 'src/keep.ts');
    console.log(`\n.cursorignore test:`);
    console.log(`Has secret.ts: ${hasSecret} (should be false — excluded by .cursorignore)`);
    console.log(`Has keep.ts: ${hasKeep} (should be true)`);

    // Verify .cursorignore is IGNORED when useIgnoreFiles is false

    for await (const f of scanFiles(tmpDir, { useIgnoreFiles: false })) {
      noIgnoreCursorFiles.push(f);

    console.log(`Has secret.ts with useIgnoreFiles:false: ${hasSecretNoIgnore} (should be true — .cursorignore disabled)`);

    // Cleanup
    rmSync(tmpDir, { recursive: true });
